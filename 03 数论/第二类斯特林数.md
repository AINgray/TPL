'''c++
	// S(n,m) 第二类斯特林数
	ll n, m;
	FIN(n, m);

	if (n - m < 0)
	{
		cout << 0 << EDL;
		return;
	}
	if (n - m == 0)
	{
		cout << 1 << EDL;
		return;
	}

	ac.Init(SZ - 1);

	mint ans = 1;
	ans *= ac.fac_inv[m];

	mint sum = 0;
	FOR(k, 0, m)
	{
		mint tmp = ac.C(m, k);
		tmp *= mint(m - k).Pow(n);
		tmp *= mint(-1).Pow(k);
		sum += tmp;
	}

	ans *= sum;
	cout << ans << EDL;
'''
